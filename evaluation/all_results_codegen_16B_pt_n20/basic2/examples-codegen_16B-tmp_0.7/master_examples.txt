// Design a module that implements an AND gate

module and_gate( 
    input a, 
    input b, 
    output out );

    // assing the AND of a and b to out
    assign          out = a & b;

    // make sure that the output is only 1 if both inputs are 1
    always_ff       @(posedge clk)
    if (a == 0 && b == 0)
        out = 0;
    else
        out = 1;
