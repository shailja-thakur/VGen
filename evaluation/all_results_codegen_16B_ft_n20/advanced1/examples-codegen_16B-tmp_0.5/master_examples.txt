    input [7:0] b,

    output overflow
);

// a signed overflow occurs if the most significant bits of a and b are low and the most significant bit of s is high
// a signed overflow may also occur if the most significant bits of a and b are high and the most significant bit of s is low


// assign the most significant bit of a to the output s
assign s = (a[7] == 0 && b[7] == 0) ? a + b : a - b;

// assign the overflow flag to 1 if the most significant bit of s is high and the most significant bit of a is low
assign overflow = (~a[7] && s[7]) || (~b[7] && ~s[7]);
