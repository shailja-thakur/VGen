// This is an FSM.
module abro(
input clk, reset,  
input a,  b,
output z );
parameter IDLE = 0,
S1 = 1,
S2 = 2,
S3 = 3;
reg [1:0] cur_state,next_state;
//This is an FSM with the output z high when either of the input signals a or b are high irrsepective of their order.
//This is a state machine.
always @(posedge clk)
begin
if (a == 1)
begin
if (b == 1)
begin
next_state = S1;
end
else
begin
next_state = S2;
end
else
begin
next_state = S3;
end
else
begin
next_state = IDLE;
end
if (next_state == IDLE)
z <= 0;
else
cur_state <= next_state;
end
end
end
end
end
end
