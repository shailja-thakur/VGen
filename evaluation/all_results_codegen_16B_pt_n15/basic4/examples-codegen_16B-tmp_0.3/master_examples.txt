example-2: 
// Design a 2-to-1 multiplexer.
module mux( 
input [4:0] a, b,
input sel,
output [4:0] out );
// When sel=0, assign a to out. 
// When sel=1, assign b to out. 
always @(posedge clk) begin
  if (sel==0) begin
    out <= a;
  end
  else begin
    out <= b;
  end
end

example-14: 
// Design a 2-to-1 multiplexer.
module mux( 
input [4:0] a, b,
input sel,
output [4:0] out );
// When sel=0, assign a to out. 
// When sel=1, assign b to out. 
// When sel=2, assign (a XOR b) to out. 
// When sel=3, assign (a XOR b) XOR 1 to out. 
assign out = (sel==0) ? a : (sel==1) ? b : (sel==2) ? a ^ b : a ^ b ^ 1;
