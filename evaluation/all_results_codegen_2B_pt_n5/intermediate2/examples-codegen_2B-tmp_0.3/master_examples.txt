// on reset, assign q to 1
// else if q is 12, assign q to 1
// else, increment q by 1 
initial begin
    q = 1;
    if ( reset == 1'b1 ) begin
        q = 1;
    end else if ( q == 12'b11 ) begin
        q = 1;
    end else begin
        q = q + 1;
    end
end
