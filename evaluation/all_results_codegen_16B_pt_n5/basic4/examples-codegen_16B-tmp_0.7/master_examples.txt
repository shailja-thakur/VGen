example-0: 
// Design a 2-to-1 multiplexer.
module mux( 
input [4:0] a, b,
input sel,
output [4:0] out );
// When sel=0, assign a to out. 
// When sel=1, assign b to out. 
assign out = a;

// When sel=0, assign a to out. 
// When sel=1, assign b to out. 
assign out = sel ? b : a;

example-3: 
// Design a 2-to-1 multiplexer.
module mux( 
input [4:0] a, b,
input sel,
output [4:0] out );
// When sel=0, assign a to out. 
// When sel=1, assign b to out. 
// When sel=2, do nothing. 
// When sel=3, assign nothing to out. 
assign out = (sel==0) ? a : b;
