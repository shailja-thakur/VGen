example-0: 
// Design a decade counter that counts from 0 through 9, inclusive.

module count_slow(
    input clk,
    input slowena,
    input reset,
    output reg [3:0] q);

// On the positive edge of the clock:
// if reset is high, reset the output q to 0. 
// Otherwise, only increment the output q if the ena input is high and q is not 9.
always @(posedge clk)
  if (reset) q <= 0;
  else if (slowena)
    if (q < 9) q <= q + 1;

example-3: 
// Design a decade counter that counts from 0 through 9, inclusive.

module count_slow(
    input clk,
    input slowena,
    input reset,
    output reg [3:0] q);

// On the positive edge of the clock:
// if reset is high, reset the output q to 0. 
// Otherwise, only increment the output q if the ena input is high and q is not 9.
// On the negative edge of the clock:
// if slowena is high, then the output q is set to 9.
// Otherwise, only decrement the output q if q is not 0.

always @(posedge clk)
    if (reset)
        q <= 0;
    else if (slowena)
        q <= 9;
    else
        q <= q + 1;

example-5: 
// Design a decade counter that counts from 0 through 9, inclusive.

module count_slow(
    input clk,
    input slowena,
    input reset,
    output reg [3:0] q);

// On the positive edge of the clock:
// if reset is high, reset the output q to 0. 
// Otherwise, only increment the output q if the ena input is high and q is not 9.
// The output q is clocked to the next digit of the decade on the negative edge of the clock.

always @(posedge clk or negedge slowena)
  if (reset)
    q <= 0;
  else if (slowena)
    if (q < 9 and slowena)
      q <= q + 1;
  end
