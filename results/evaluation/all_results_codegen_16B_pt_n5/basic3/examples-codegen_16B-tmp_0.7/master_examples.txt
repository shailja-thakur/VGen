// This is a 3-bit priority encoder. If none of the input bits are high, output is zero.
module priority_encoder( 
input [2:0] in,
output reg [1:0] pos ); 
// When sel=1, assign b to out
// When sel=0, assign a to out
always @(posedge clk) begin
	if( sel==1 )
		pos <= 0;
	else
		pos <= in[0];
end

example-2: 
// This is a 3-bit priority encoder. If none of the input bits are high, output is zero.
module priority_encoder( 
input [2:0] in,
output reg [1:0] pos ); 
// When sel=1, assign b to out
// When sel=0, assign a to out
// When sel=2, assign a&b to out
always @(posedge clk) begin
    case(sel)
        case(in)
            pos <= in;
        endcase
    endcase

// This is a 3-bit priority encoder. If none of the input bits are high, output is zero.
module priority_encoder( 
input [2:0] in,
output reg [1:0] pos ); 
// When sel=1, assign b to out
// When sel=0, assign a to out
always @( posedge in) begin
  if( in[0] | in[1] ) pos <= 0;
  else pos <= 1;
end

example-4: 
// This is a 3-bit priority encoder. If none of the input bits are high, output is zero.
module priority_encoder( 
input [2:0] in,
output reg [1:0] pos ); 
// When sel=1, assign b to out
reg [1:0] sel = 0;
// When r=1, assign carry to pos
reg [1:0] r = 0;
always @(posedge clk) 
    if(sel==1) 
        pos <= in;
    else
        pos <= r;
end
