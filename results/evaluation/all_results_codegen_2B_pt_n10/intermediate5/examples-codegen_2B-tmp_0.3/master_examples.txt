// left shift and rotate operation
module left_rotate(clk,reset,amount,data,load,out);
input clk,reset;
input [2:0] amount;
input [7:0] data;
input load;
output reg [7:0] out;
// when load is high, load data to out
// otherwise rotate the out register followed by left shift the out register by amount bits
always @(posedge clk)
  if (load)
  {
    out = data;
  }
  else
  {
    out <<= amount;
    out = out >> (32 - amount);
  }

// left shift and rotate operation
module left_rotate(clk,reset,amount,data,load,out);
input clk,reset;
input [2:0] amount;
input [7:0] data;
input load;
output reg [7:0] out;
// when load is high, load data to out
// otherwise rotate the out register followed by left shift the out register by amount bits
always @(posedge clk)
  if (reset) begin
    out <= 0;
  end
  else if (load) begin
    out <= data;
  end
  else begin
    // rotate the out register by amount bits
    out <= (data << amount);
    // left shift the out register by amount bits
    out <= (out >> amount);
  end

// left shift and rotate operation
module left_rotate(clk,reset,amount,data,load,out);
input clk,reset;
input [2:0] amount;
input [7:0] data;
input load;
output reg [7:0] out;
// when load is high, load data to out
// otherwise rotate the out register followed by left shift the out register by amount bits
always @(posedge clk,posedge reset)
begin
    if (reset) begin
        out <= 0;
    end else if (load) begin
        out <= data;
    end else begin
        out <= (data >> amount);
        out <<= amount;
    end
end

// left shift and rotate operation
module left_rotate(clk,reset,amount,data,load,out);
input clk,reset;
input [2:0] amount;
input [7:0] data;
input load;
output reg [7:0] out;
// when load is high, load data to out
// otherwise rotate the out register followed by left shift the out register by amount bits
always @(posedge clk)
    if (load)
        out = data;
    else if (amount == 0)
        out = data;
    else
    {
        out = data;
        for (int i = 0; i < amount; i++)
            out = data << 1;
    }
