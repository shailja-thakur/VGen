// Design a decade counter that counts from 0 through 9, inclusive.

module count_slow(
    input clk,
    input slowena,
    input reset,
    output reg [3:0] q);

// On the positive edge of the clock:
// if reset is high, reset the output q to 0. 
// Otherwise, only increment the output q if the ena input is high and q is not 9.
always @(posedge clk)
  if (reset)
    q <= 0;
  else if (slowena)
    if (q < 9)
      q <= q + 1;

// Design a decade counter that counts from 0 through 9, inclusive.

module count_slow(
    input clk,
    input slowena,
    input reset,
    output reg [3:0] q);

// On the positive edge of the clock:
// if reset is high, reset the output q to 0. 
// Otherwise, only increment the output q if the ena input is high and q is not 9.
always @(posedge clk)
  if (reset)
    q <= 0;
  else if (slowena)
    if (q < 9)
      q <= q + 1;

// On the negative edge of the clock:
// if slowena is high, decrement the output q if it is not 0. 
// Otherwise, only decrement the output q if q is not 0.
always @(negedge clk)
  if (slowena)
    if (q > 0)
      q <= q - 1;
  else if (q > 0)
    q <= q - 1;

module count_slow(
    input clk,
    input slowena,
    input reset,
    output reg [3:0] q);

// On the positive edge of the clock:
// if reset is high, reset the output q to 0. 
// Otherwise, only increment the output q if the ena input is high and q is not 9.
always @(posedge clk)
begin
    if (reset) begin
        q <= 0;
    end
    else if (slowena) begin
        if (q < 9) begin
            q <= q + 1;
        end
    end
